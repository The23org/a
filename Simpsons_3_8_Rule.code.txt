%Simpson 3/8

function simpson_method()
x1=input('Enter lower limit of x1=');
xn=input('Enter upper limit xn=');
n=input('Enter number of strips n=');
h=(xn-x1)/n;
area=0;
for i=1:1:n
 area=area+(3*h/8)*(f(x1)+3*f(x1+h)+3*f(x1+2*h)+f(x1+3*h));
 x1=x1+h;
end
fprintf('Area=%f',area)
end
function y=f(x)
y= (exp(-x^2))/(1+x);
end
